    for(int i=0; i<7; i++){

            maior_chuva = chuvas[i];
        }

            menor_chuva = chuvas[i];

    }

}

#include <iostream>
#include <cstdlib>
#include <ctime>

int main() {

    //Declarando as variaveis
	int megasena[6] = {};
	int megasena_jogada[6] = {};
	int jogada=0, numero_acertos=0, n_mega=0;
	bool repetido=false;

    //while para repetir todo o menu para que os numeros sejam validos
	int menu=0;
	while(menu < 6) {
		system("clear");
		//Interacao com o usuario
		std::cout << "\n//////////////////////";
		std::cout << "\n       MEGASENA       ";
		std::cout << "\n//////////////////////";
		std::cout << "\nInsira o " << menu+1 << " numero da MegaSena: ";
		std::cin >> jogada;

        //Para que o numero nao seja repetido e
        //para que o numero seja maior que zero e menor que 61
		for(int i=0; i<6; i++) {
			if(jogada==megasena_jogada[i]) {
				repetido=true;
			}
		}
		if(repetido==false) {
			if(jogada>0 && jogada<61) {
				megasena_jogada[menu] = jogada;
				menu++;
			} else {
				system("clear");
				std::cout << "Insira outro numero! de 1 a 60 e sem repeticoes!";
			}
		} else if(repetido==true) {
			system("clear");
			std::cout << "Insira outro numero! de 1 a 60 e sem repeticoes!"  ;
			repetido=false;
		}
	}
    
    //Gerando os numeros aleatorios
    //E garantindo que eles nao se repitam
	srand(time(0));

	int num_rand=0;
	while(num_rand < 6) {
		n_mega = rand()%61;
		for(int j=0; j<6; j++) {
			if(n_mega==megasena[j]) {
				repetido=true;
			}
		}
		if(repetido==false) {
			megasena[num_rand] = n_mega;
			num_rand++;
		} else if(repetido==true) {
			repetido=false;
		}
	}
	//Gerando os numeros aleatorios

	//calculo de numero de acertos
		for(int j=0; j<6; j++) {
			for(int i=0; i<6; i++) {
				if(megasena_jogada[i]==megasena[j]) {
					numero_acertos++;
				}
			}
		}

    //limpeza
	system("clear");

    //Menu para a quantidade de acertos
	if(numero_acertos==6) {
		std::cout << "Parabens! Voce acertou todos os numeros e venceu a megasena!" << std::endl;
		std::cout << "Quantidade de acertos: " << numero_acertos;
	} else if(numero_acertos==5) {
		std::cout << "Parabens! Voce acertou a quina da megasena!" << std::endl;
		std::cout << "Quantidade de acertos: " << numero_acertos;
	} else if(numero_acertos==4) {
		std::cout << "Parabens! Voce acertou a quadra da megasena!" << std::endl;
		std::cout << "Quantidade de acertos: " << numero_acertos;
	} else if(numero_acertos<4) {
		std::cout << "Nao foi dessa vez!" << std::endl;
		std::cout << "Quantidade de acertos: " << numero_acertos;
	}

    //Jogadas do jogador e Megasena
	std::cout << "\n\nSuas jogadas: " << std::endl;
	for(int i=0; i<6; i++) {
		std::cout << megasena_jogada[i] << " ";
	}
	std::cout << "\n\nMegaSena: " << std::endl;
	for(int i=0; i<6; i++) {
		std::cout << megasena[i] << " ";
	}

    //Fim
	return 0;
}
